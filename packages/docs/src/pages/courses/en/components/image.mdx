import { Image, CodeExample, getExampleImage } from '@coffic/cosy-ui';

## Basic Usage

The simplest way to display images

<CodeExample
  code={`<Image src={getExampleImage({ width: 400, height: 300 })} alt="Example image" />`}
>
  <Image src={getExampleImage({ width: 400, height: 300 })} alt="Example image" />
</CodeExample>

## Rounded Corners and Shadow Effects

Add visual effects using the `rounded` and `shadow` properties

<CodeExample
  code={`<Image src={getExampleImage({ width: 400, height: 300, randomSeed: 1 })} alt="Example image" rounded="lg" shadow="md" />`}
>
  <Image src={getExampleImage({ width: 400, height: 300, randomSeed: 1 })} alt="Example image" rounded="lg" shadow="md" />
</CodeExample>

## Hover Interaction Effects

Add hover effects using the `hover` property, supporting scaling, brightness changes, and blur effects

<CodeExample
  code={`<Image src={getExampleImage({ width: 400, height: 300, randomSeed: 2 })} alt="Example image" hover="scale" rounded="md" />`}
>
  <Image src={getExampleImage({ width: 400, height: 300, randomSeed: 2 })} alt="Example image" hover="scale" rounded="md" />
</CodeExample>

## Loading Indicators

Display different types of loading indicators when remote images are loading

<CodeExample
  code={`<Image src={getExampleImage({ width: 800, height: 600, randomSeed: 3 })} alt="Large remote image" loadingIndicator="spinner" />`}
>
  <Image src={getExampleImage({ width: 800, height: 600, randomSeed: 3 })} alt="Large remote image" loadingIndicator="spinner" />
</CodeExample>

<CodeExample
  code={`<Image src={getExampleImage({ width: 800, height: 600, randomSeed: 4 })} alt="Large remote image" loadingIndicator="progress" />`}
>
  <Image src={getExampleImage({ width: 800, height: 600, randomSeed: 4 })} alt="Large remote image" loadingIndicator="progress" />
</CodeExample>

## Object Fit Methods

Control how images fill their containers using the `objectFit` property

<CodeExample
  code={`<Image src={getExampleImage({ width: 600, height: 400, randomSeed: 5 })} alt="Example image" objectFit="contain" width={300} height={200} />`}
>
  <Image src={getExampleImage({ width: 600, height: 400, randomSeed: 5 })} alt="Example image" objectFit="contain" width={300} height={200} />
</CodeExample>

## Transition Animation Effects

Add transition animation effects using the `transition` property

<CodeExample
  code={`<Image src={getExampleImage({ width: 400, height: 300, randomSeed: 6 })} alt="Example image" transition="fade" hover="brightness" />`}
>
  <Image src={getExampleImage({ width: 400, height: 300, randomSeed: 6 })} alt="Example image" transition="fade" hover="brightness" />
