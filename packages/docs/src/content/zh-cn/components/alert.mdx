import { Alert, CodeExample } from '@coffic/cosy-ui';

## 基础提示

最简单的提示信息展示

<CodeExample
  code={`<Alert type="info">这是一条信息提示</Alert>`}
>
  <Alert type="info">这是一条信息提示</Alert>
</CodeExample>

## 带标题的提示

通过 title 属性添加标题，使提示信息更加醒目

<CodeExample
  code={`<Alert type="success" title="操作成功">您的操作已成功完成</Alert>`}
>
  <Alert type="success" title="操作成功">您的操作已成功完成</Alert>
</CodeExample>
