import { Banner, CodeExample } from '@/index';

## Basic Banner

The simplest banner information display, using the theme color background by default

<CodeExample
  code={`<Banner>Welcome to our service</Banner>`}
>
  <Banner>Welcome to our service</Banner>
</CodeExample>

## Different Background Colors

Set different background colors using the bgColor property

<CodeExample
  code={`<Banner bgColor="secondary">Explore more features</Banner>`}
>
  <Banner bgColor="secondary">Explore more features</Banner>
</CodeExample>

<CodeExample
  code={`<Banner bgColor="success">Operation completed successfully</Banner>`}
>
  <Banner bgColor="success">Operation completed successfully</Banner>
</CodeExample>

## Custom Styles

Add custom style classes using the class property

<CodeExample
  code={`<Banner class="cosy:text-4xl cosy:py-10">Building beautiful digital experiences</Banner>`}
>
  <Banner class="cosy:text-4xl cosy:py-10">Building beautiful digital experiences</Banner>
</CodeExample>

## Disable Animation Effects

Control whether animation effects are enabled using the animated property

<CodeExample
  code={`<Banner animated={false}>Banner without animation effects</Banner>`}
>
  <Banner animated={false}>Banner without animation effects</Banner>
</CodeExample>

## Using in Page Layouts

The Banner component can flexibly integrate into various page layouts

```astro
<header>
  <nav>...</nav>
  <Banner>Special offer: Limited time discount</Banner>
</header>

<main>
  <section>
    <Banner bgColor="accent">Learn about our latest products</Banner>
    <!-- Content -->
  </section>
</main>

<footer>
  <Banner bgColor="secondary">Thank you for your visit</Banner>
</footer>
```
