import { Link, CodeExample } from '@coffic/cosy-ui';

## 基础用法

最简单的链接展示方式

<CodeExample code={`<Link href="/about">关于我们</Link>`}>
  <Link href="/about">关于我们</Link>
</CodeExample>

## 外部链接

链接到外部网站，自动添加新窗口打开属性

<CodeExample
  code={`<Link href="https://example.com" external>访问外部网站</Link>`}
>
  <Link href="https://example.com" external>
    访问外部网站
  </Link>
</CodeExample>

## 样式变体

通过`variant`属性应用不同的样式变体

<CodeExample
  code={`<Link href="/signup" variant="primary">主要链接</Link>
<Link href="/learn-more" variant="secondary">次要链接</Link>
<Link href="/docs" variant="text">文本链接</Link>
<Link href="/get-started" variant="cta">行动号召</Link>
<Link href="/preview" variant="ghost">幽灵链接</Link>`}
>
  <Link href="/signup" variant="primary">
    主要链接
  </Link>
  <Link href="/learn-more" variant="secondary" class="cosy:ml-2">
    次要链接
  </Link>
  <Link href="/docs" variant="text" class="cosy:ml-2">
    文本链接
  </Link>
  <Link href="/get-started" variant="cta" class="cosy:ml-2">
    行动号召
  </Link>
  <Link href="/preview" variant="ghost" class="cosy:ml-2">
    幽灵链接
  </Link>
</CodeExample>

## 块级显示

通过`block`属性使链接占据整行显示

<CodeExample
  code={`<Link href="/contact" block variant="primary">独占一行的链接</Link>
<p class="cosy:mt-2">这是一段文字，其中包含一个<Link href="/inline">内联链接</Link>。</p>`}
>
  <Link href="/contact" block variant="primary">
    独占一行的链接
  </Link>
  <p class="cosy:mt-2">
    这是一段文字，其中包含一个<Link href="/inline">内联链接</Link>。
  </p>
</CodeExample>

## 动画效果

通过`animation`属性添加悬停动画效果

<CodeExample
  code={`<Link href="/contact" animation="hover-lift" variant="primary">上浮效果</Link>
<Link href="/features" animation="hover-glow" variant="secondary">发光效果</Link>
<Link href="/pricing" animation="hover-scale" variant="cta">缩放效果</Link>`}
>
  <Link href="/contact" animation="hover-lift" variant="primary">
    上浮效果
  </Link>
  <Link
    href="/features"
    animation="hover-glow"
    variant="secondary"
    class="cosy:ml-2"
  >
    发光效果
  </Link>
  <Link href="/pricing" animation="hover-scale" variant="cta" class="cosy:ml-2">
    缩放效果
  </Link>
</CodeExample>

## 尺寸变体

通过`size`属性控制链接的尺寸

<CodeExample
  code={`<Link href="/small" size="sm">小型链接</Link>
<Link href="/medium" size="md">中型链接</Link>
<Link href="/large" size="lg">大型链接</Link>`}
>
  <Link href="/small" size="sm">
    小型链接
  </Link>
  <Link href="/medium" size="md" class="cosy:ml-2">
    中型链接
  </Link>
  <Link href="/large" size="lg" class="cosy:ml-2">
    大型链接
  </Link>
</CodeExample>

## 文本居中

通过`centerText`属性使链接文本居中显示

<CodeExample
  code={`<Link href="/centered" block centerText variant="primary">居中显示的链接文本</Link>`}
>
  <Link href="/centered" block centerText variant="primary">
    居中显示的链接文本
  </Link>
</CodeExample>

## 调试模式

通过`debug`属性显示链接边界，便于开发调试

<CodeExample code={`<Link href="/debug" debug>显示边界的链接</Link>`}>
  <Link href="/debug" debug>
    显示边界的链接
  </Link>
</CodeExample>
