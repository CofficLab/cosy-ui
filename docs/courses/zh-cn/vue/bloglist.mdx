import { CodeExample } from '@cosy/index';
import { BlogList, BlogListExampleCodes, BlogListExamples } from '@cosy/index_vue';

## 基础用法

基本的博客文章列表展示，支持中文界面。

<CodeExample
  code={BlogListExampleCodes.Basic}
>
  <BlogListExamples.Basic client:load />
</CodeExample>

## 空状态展示

当没有博客文章时，组件会显示一个友好的空状态提示。

<CodeExample
  code={BlogListExampleCodes.Empty}
>
  <BlogListExamples.Empty client:load />
</CodeExample>

## 英文界面

组件支持英文界面，只需设置 `lang="en"` 即可。

<CodeExample
  code={BlogListExampleCodes.English}
>
  <BlogListExamples.English client:load />
</CodeExample>

## 英文空状态

英文环境下的空状态展示。

<CodeExample
  code={BlogListExampleCodes.EmptyEnglish}
>
  <BlogListExamples.EmptyEnglish client:load />
</CodeExample>

## 属性

| 属性名 | 类型 | 默认值 | 说明 |
| --- | --- | --- | --- |
| blogs | IBlog[] | - | 博客文章数据数组 |
| lang | 'zh' \| 'en' | 'zh' | 语言设置，影响空状态展示文本 |

## 类型定义

```ts
interface IBlog {
    id: string;       // 博客文章唯一标识
    title: string;    // 博客文章标题
    link: string;     // 博客文章链接
    tags: string[];   // 博客文章标签
}
```
