import { Alert, CodeExample } from '@cosy/index';

## Basic Alert

The simplest way to display alert information

<CodeExample
  code={`<Alert type="info">This is an information alert</Alert>`}
>
  <Alert type="info">This is an information alert</Alert>
</CodeExample>

## Alert with Title

Add a title using the title attribute to make the alert more prominent

<CodeExample
  code={`<Alert type="success" title="Operation Successful">Your operation has been completed successfully</Alert>`}
>
  <Alert type="success" title="Operation Successful">Your operation has been completed successfully</Alert>
</CodeExample>
