import { Card, CodeExample, getProductImage, getExampleImage } from '@cosy/index';

## Basic Card

The simplest card display, including title and description

<CodeExample
  code={`<Card 
  title="Card Title" 
  subtitle="Card description text"
>
  Here is the card's other content
</Card>`}
>
  <Card 
    title="Card Title" 
    subtitle="Card description text"
  >
    Here is the card's other content
  </Card>
</CodeExample>

## Card with Image

Add images via the imageUrl property to enhance visual appeal

<CodeExample
  code={`<Card 
  title="Product Name" 
  subtitle="Product Description" 
  imageUrl={getProductImage({ width: 400, height: 200 })}
>
  Other product information
</Card>`}
>
  <Card 
    title="Product Name" 
    subtitle="Product Description" 
    imageUrl={getProductImage({ width: 400, height: 200 })}
  >
    Other product information
  </Card>
</CodeExample>

## Clickable Card

Make the entire card a link using the href property

<CodeExample
  code={`<Card 
  title="Article Title" 
  subtitle="Article Summary" 
  imageUrl={getExampleImage({ width: 400, height: 200, tag: 'article' })} 
  href="#"
/>`}
>
  <Card 
    title="Article Title" 
    subtitle="Article Summary" 
    imageUrl={getExampleImage({ width: 400, height: 200, tag: 'article' })} 
    href="#"
  />
</CodeExample>

## Custom Styles

Add custom style classes using the class property

<CodeExample
  code={`<Card 
  title="Custom Card" 
  subtitle="Using custom styles" 
  class="cosy:bg-primary cosy:text-white"
>
  Card content will inherit the theme color and white text
</Card>`}
>
  <Card 
    title="Custom Card" 
    subtitle="Using custom styles" 
    class="cosy:bg-primary cosy:text-white"
  >
    Card content will inherit the theme color and white text
  </Card>
</CodeExample>

## Using in Page Layouts

Card components can be flexibly used in various page layouts

```astro
<div class="cosy:grid cosy:grid-cols-1 cosy:md:grid-cols-2 cosy:lg:grid-cols-3 cosy:gap-6">
  <Card 
    title="Feature One" 
    subtitle="Feature description" 
    imageUrl={getExampleImage({ width: 400, height: 200, tag: 'feature', randomSeed: 1 })}
  />
  <Card 
    title="Feature Two" 
    subtitle="Feature description" 
    imageUrl={getExampleImage({ width: 400, height: 200, tag: 'feature', randomSeed: 2 })}
  />
  <Card 
    title="Feature Three" 
    subtitle="Feature description" 
    imageUrl={getExampleImage({ width: 400, height: 200, tag: 'feature', randomSeed: 3 })}
  />
</div>
```

## Card Groups

Combine multiple cards to display related content

```astro
<section>
  <h2>Our Products</h2>
  <div class="cosy:flex cosy:flex-wrap cosy:gap-4">
    <Card 
      title="Product One" 
      subtitle="Product description" 
      imageUrl={getProductImage({ randomSeed: 'product1' })} 
      href="/products/product1"
    />
    <Card 
      title="Product Two" 
      subtitle="Product description" 
      imageUrl={getProductImage({ randomSeed: 'product2' })} 
      href="/products/product2"
    />
  </div>
</section>
```
